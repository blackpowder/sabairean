កែប្រែល្បែងកំសាន្តអោយក្លាយទៅជារបស់អ្នក

នេះជាពេលដែលកែប្រែល្បែងកំសាន្តនេះអោយក្លាយទៅជារបស់អ្នក ។ អ្នកអាចបង្កើត .apk file បានច្រើនទៅតាមតំរូវការរបស់អ្នក ហើយអាចដាក់ឈ្មោះរបស់អ្នកទៅកាន់ផ្ទាំងស្វាគមន៍ ដោយភ្ជាប់មកជាមួយនឹងរូបថតរបស់អ្នកផ្ទាល់ ។

សូមអនុវត្តន៍តាមការបង្ហាញខាងក្រោមនេះតាមលំដាប់លំដោយដើម្បីធ្វើការកែប្រែ

បង្ហាញឈ្មោះរបស់លោកអ្នកនៅលើផ្ទាំងស្វាគមន៍

  • បើក Eclipse ហើយចូលទៅកាន់ Project Workspace
  • ដោយសារតែ Project ឈ្មោះ MemoryGame ដូច្នេះសូមចូលទៅកាន់ Folder ដូចខាងក្រោម
  • MemoryGame -> res -> values -> string.xml
  • នៅក្នុងរង្វង់ព៌ណក្រហម កូពីខ្មែរយូនីកូដដែលអ្នកបានធ្វើការបំលែងនៅក្នុងជំពូក Converting Khmer Unicode
  • នៅពេលរួចរាល់សូមចុច Ctrl + S ដើម្បីរក្សាទុកការងាររបស់អ្នក
  • ត្រង់ចំនុចនេះលោកអ្នកក៏អាចបង្កើត .apk file ដោយអនុវត្តន៍ទៅតាមជំពូក (Deploy Android App -> Create APK File) រឺក៏បង្ហាញនៅលើ Emulator ដោយអនុវត្តន៍ទៅតាមជំពូក (Making An Android Project -> Step 4)

  • បើក Eclipse ហើយចូលទៅកាន់ Project​ Workspace

  • MemoryGame -> res -> layout -> main.xml
  • នៅលើ Keyboard របស់អ្នកចុច Ctrl + F ដើម្បីស្វែងរកពាក្យ TableRow03
  • នៅក្នុងរង្វង់ព៌ណក្រហមជំនួសពាក្យ card01 ដោយឈ្មោះរូបថតរបស់អ្នក
  • នៅពេលរួចរាល់សូមចុច Ctrl + S ដើម្បីរក្សាទុកការងាររបស់អ្នក
